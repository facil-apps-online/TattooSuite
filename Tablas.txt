create table public.attentions (
  id uuid not null default gen_random_uuid (),
  client_id uuid not null,
  status text not null default 'Confirmada'::text,
  notes text null,
  total_amount numeric not null default 0,
  created_at timestamp with time zone not null default now(),
  updated_at timestamp with time zone not null default now(),
  tenant_id uuid not null,
  branch_id uuid not null,
  attention_datetime timestamp with time zone null,
  constraint attentions_pkey primary key (id),
  constraint attentions_client_id_fkey foreign KEY (client_id) references clients (id) on delete CASCADE,
  constraint fk_attentions_branch foreign KEY (branch_id) references branches (id) on delete CASCADE,
  constraint fk_attentions_tenant foreign KEY (tenant_id) references tenants (id) on delete CASCADE,
  constraint attentions_status_check check (
    (
      status = any (
        array[
          'Confirmada'::text,
          'En Proceso'::text,
          'Finalizada'::text,
          'Pagada'::text,
          'Cancelada'::text
        ]
      )
    )
  )
) TABLESPACE pg_default;

create trigger audit_attentions_changes
after INSERT
or DELETE
or

execute FUNCTION audit_trigger_function ();

create trigger update_attentions_updated_at BEFORE

execute FUNCTION update_updated_at_column ();
